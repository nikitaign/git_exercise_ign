import java.util.Scanner;

public class ConverterNew {
        public static void main(String[] args) {
            final double ROUBLES_PER_DOLLAR = 92.12;

            Scanner scanner = new Scanner(System.in);
            System.out.print("Введите сумму в долларах: ");
            int dollars = scanner.nextInt();

            int lastDigit = dollars % 10;

            if (dollars >=5 && dollars <=20)
                System.out.println(dollars + " американских долларов");
                else if (lastDigit == 1)
                    System.out.println(dollars + " американский доллар");
                else if (lastDigit >= 2 && lastDigit <=4)
                    System.out.println(dollars + " американских доллара");
                    else System.out.println(dollars + " американских долларов");

            double roubles = ROUBLES_PER_DOLLAR * dollars;
            System.out.println("Сумма в российских рублях: " + roubles);

        }
}
